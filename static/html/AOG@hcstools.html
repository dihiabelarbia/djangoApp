#!/usr/bin/env python
# -*- coding: utf-8 -*-
#

import subprocess
import sys
import os

def lectureTab(opWD, key, logwrite=None):
    """
    Recuperation des parametres du modele
    """
    filename = os.path.join(opWD, "common", "tablei", "applis.tab")
    paramtab = {}
    # Chargement de la table
    logwrite.info("  Lecture du fichier : " + filename)
    # self.logwrite.info("Lecture '" + filename + "' pour récupération du paramètrage" )
    try:
        # self.logwrite.info(filename)
        fic = open(filename, "r")
    except Exception as ex:
        logwrite.error( "Probleme d'ouverture en lecture de la table '" + filename)
        raise
    #
    # Lecture ligne a ligne de la table avec recherche du template (modele)
    found = False
    try:
        for line in fic:
            tab = line.strip("\n").split("\t")
            if (tab[0].upper() == key.upper()):
                # paramtab["dataloader"] = tab[1]
                # paramtab["dataloaderversion"] = tab[2]
                # paramtab["tag"] = tab[3]
                # paramtab["tagversion"] = tab[4]

                paramtab["resdescid"]  = tab[1]
                paramtab["opAppli"]    = tab[3]
                paramtab["opFam"]      = tab[4]
                paramtab["dataloader"] = tab[5]

                found = True
                break
    except Exception as ex:
        logwrite.error( "Probleme de lecture de la table '" + filename + "' avec la cle '" + key)
        raise
    #
    # Fermeture du fichier
    fic.close()
    #
    # Test si modele trouve dans la table
    if found is False :
        logwrite.error( "Absence du modele '" + key + "' dans la table '" + filename + "'")
    #
    # Ecriture dans la trace des parametres trouves
    logwrite.debug("Parametres de modeles trouves : " + str(paramtab) )
    #
    return paramtab


def runCommand(commandline):
    """
    Fonction de lancement d'une commande
    """
    stdoutdata = subprocess.PIPE
    stderrdata = subprocess.PIPE

    my_env = os.environ.copy()

    if (os.name == 'posix'):
        process = subprocess.Popen(
            commandline, shell=True, stdout=stdoutdata, stderr=stderrdata)
    else:
        process = subprocess.Popen(
            commandline, stdout=stdoutdata, stderr=stderrdata)
    (stdOut, stdErr) = process.communicate()
    exitCode = process.wait()
    pid = process.pid
    return exitCode, pid, stdOut, stdErr

def getNbPagesInd(pathfile):
    with open(pathfile, encoding='utf-8') as indfile:
        text = indfile.readlines()
        return len(text)
