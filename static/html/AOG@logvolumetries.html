#!python3
# -*- coding: utf-8 -*-
import os
import openpyxl
import sys
import datetime
import shutil

months = ["janvier", "février", "mars", "avril", "mai", "juin", "juillet", "aout", "septembre", "octobre", "novembre", "décembre"]

class volumetries():
    def __init__(self, onglet, logwrite=None, env=False):
        if logwrite is None:
            raise Exception("Aucun fichier de log défini dans l'écriture des volumétries : " + __file__)

        self.logwrite = logwrite
        if env == "PROD":
            self.xlsxfile = "O:\\5- SUIVI Editique CCM\\0-Budget\\VOLUMETRIES\\VOLUMETRIES.xlsx"
        elif env == "DEV":
            self.xlsxfile = "O:\\5- SUIVI Editique CCM\\0-Budget\\VOLUMETRIES\\VOLUMETRIES_DEV.xlsx"
        elif env == "PPROD":
            self.xlsxfile = "O:\\5- SUIVI Editique CCM\\0-Budget\\VOLUMETRIES\\VOLUMETRIES_PPROD.xlsx"
        else:
            self.xlsxfile = "O:\\5- SUIVI Editique CCM\\0-Budget\\VOLUMETRIES\\VOLUMETRIES_TESTS.xlsx"
        now = datetime.datetime.today()
        timestamp = now.strftime("%Y%m%d%H%M%S")
        self.svgfile = self.xlsxfile + "_" + timestamp + ".svg"
        shutil.copy(self.xlsxfile, self.svgfile)
        self.logwrite.info("Sauvegarde du fichier " + self.xlsxfile + " en " + self.svgfile)

        self.tab = onglet
        self.openxlsxfile()
        self.nbvaluesadded = 0

    def openxlsxfile(self):
        if os.path.isfile(self.xlsxfile) is False:
            self.logwrite.error("Le fichier " + self.xlsxfile + " n'est pas accessible")
            raise

        try:
            self.wb = openpyxl.load_workbook(self.xlsxfile)
        except Exception as ex:
            self.logwrite.error("Le fichier " + self.xlsxfile + " n'est pas accessible")
            raise
        self.logwrite.info("Ouverture du fichier " + self.xlsxfile)
        try:
            self.ws = self.wb[self.tab]
        except Exception as ex:
            self.logwrite.error("L'onglet " + self.tab + " n'est pas accessible")
            raise
        self.logwrite.info("Chargement de l'onglet " + self.tab)

    def startarray(self, col, row, year):
        self.coldeb = col
        self.rowdeb = row
        self.yeardeb = year
        self.logwrite.debug("La case de départ est col: "  + str(col) + " / row " + str(row))
        self.logwrite.debug("L'année de départ est : " + str(year))

    def setvalue(self, year, month, day, nbenvoisposte):
        rowval = month + self.rowdeb + ((year - self.yeardeb) * 12) - 1
        colval = day + self.coldeb - 1
        self.ws.cell(row=rowval, column=colval).value = int(nbenvoisposte)
        self.nbvaluesadded = self.nbvaluesadded + 1
        if self.nbvaluesadded == 15:
            self._savearray()
            self.nbvaluesadded = 0
        self.logwrite.debug("dans la colonne "  + str(colval) + " / ligne " + str(rowval))

        self.logwrite.info("Volumétrie de " + str(nbenvoisposte) + " plis en date du " + str(day).zfill(2) + " " + str(months[month - 1]) + " " + str(year))

    def _savearray(self):
        try:
            self.wb.save(self.xlsxfile)
            self.logwrite.info("Enregistrement du fichier " + self.xlsxfile)
        except Exception as ex:
            self.logwrite.error("Enregistrement du fichier " + self.xlsxfile + " impossible")
            raise

    def closefile(self):
        self._savearray()
        os.remove(self.svgfile)
        self.logwrite.info("Suppression du fichier de backup")
